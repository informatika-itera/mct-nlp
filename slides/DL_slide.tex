\begin{frame}
    \frametitle{Mengapa Deep Learning?}
    \framesubtitle{Keterbatasan yang membuat kita perlu naik level}
    \small
    Pada sesi-sesi sebelumnya kita sudah pakai \textbf{TF-IDF + SVM / PyCaret}.\\[0.4em]
    Ternyata ada beberapa hal yang \tem{tidak bisa} dilakukan pendekatan itu:
    \vspace{0.3em}
    \begin{itemize}
        \item \textbf{Tidak peduli urutan kata} — "saya \textit{tidak} suka" vs "saya suka \textit{tidak} ada masalah" dianggap mirip.
        \item \textbf{Tidak paham konteks} — kata "bank" di kalimat "tepi sungai" vs "menabung di bank" dianggap sama.
        \item \textbf{Dimensi fitur meledak} — setiap kata unik menjadi satu kolom; ribuan kata = ribuan kolom.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Mengapa Deep Learning?}
    \framesubtitle{Apa yang ditawarkan Deep Learning}
    \small
    \textit{Deep Learning} menyelesaikan ketiga masalah tadi dengan cara:
    \vspace{0.3em}
    \begin{itemize}
        \item \textbf{Belajar representasi sendiri} — model secara otomatis menemukan pola penting dari data, tanpa kita perlu rekayasa fitur manual.
        \item \textbf{Memori urutan} — arsitektur RNN / LSTM "membaca" kata satu per satu sambil mengingat apa yang sudah dibaca.
        \item \textbf{Konteks global} — mekanisme \textit{attention} memungkinkan model melihat seluruh kalimat sekaligus.
    \end{itemize}
    \begin{exampleblock}{Analogi Sederhana}
        Kalau TF-IDF seperti \textbf{kamus kata} yang hanya menghitung frekuensi, maka DL seperti \textbf{orang yang membaca} — ia mengerti konteks dan urutan.
    \end{exampleblock}
\end{frame}

\begin{frame}
    \frametitle{TF-IDF: Contoh Representasi}
    \framesubtitle{Tiga dokumen, satu matriks fitur}
    \small
    \textbf{Corpus:}
    \quad \texttt{D1: "saya suka kopi"} \quad
    \texttt{D2: "saya suka teh"} \quad
    \texttt{D3: "kopi enak sekali"}
    \vspace{0.4em}

    \textbf{Vocabulary} (urutan abjad): \texttt{enak | kopi | saya | sekali | suka | teh}
    \vspace{0.4em}

    \begin{center}
    \small
    \setlength{\tabcolsep}{5pt}
    \begin{tabular}{l|cccccc}
        \toprule
         & \texttt{enak} & \texttt{kopi} & \texttt{saya} & \texttt{sekali} & \texttt{suka} & \texttt{teh} \\
        \midrule
        D1 & 0 & \textbf{0.58} & \textbf{0.58} & 0 & \textbf{0.58} & 0 \\
        D2 & 0 & 0 & \textbf{0.58} & 0 & \textbf{0.58} & \textbf{0.58} \\
        D3 & \textbf{0.58} & \textbf{0.58} & 0 & \textbf{0.58} & 0 & 0 \\
        \bottomrule
    \end{tabular}
    \end{center}
    \vspace{0.3em}
    \begin{itemize}
        \item Setiap \textbf{baris} = satu dokumen; setiap \textbf{kolom} = satu kata.
        \item Nilai $> 0$ hanya untuk kata yang \textit{muncul} di dokumen tersebut.
        \item D1 dan D3 sama-sama punya kata \texttt{kopi} $\Rightarrow$ ada sedikit kemiripan.
        \item Tapi \texttt{suka} dan \texttt{enak} yang bermakna mirip? \textbf{Tidak terhubung sama sekali.}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Kelemahan TF-IDF \#1}
    \framesubtitle{Tidak Paham Makna (Semantik)}
    \small
    TF-IDF hanya melihat \textbf{kata yang persis sama} --- bukan maknanya.
    \vspace{0.5em}

    \begin{columns}[T]
        \column{0.47\textwidth}
        \begin{tcolorbox}[colback=gray!10, colframe=gray!50, title=Kalimat A, fonttitle=\small\bfseries]
            \texttt{mobil mogok di jalan}
        \end{tcolorbox}
        \column{0.47\textwidth}
        \begin{tcolorbox}[colback=gray!10, colframe=gray!50, title=Kalimat B, fonttitle=\small\bfseries]
            \texttt{kendaraan rusak di jalanan}
        \end{tcolorbox}
    \end{columns}
    \vspace{0.5em}
    \begin{alertblock}{Masalah}
        Tidak ada satu pun kata yang sama $\Rightarrow$ TF-IDF menganggap kedua kalimat \textbf{tidak berkaitan sama sekali}, padahal maknanya hampir identik.
    \end{alertblock}
    \vspace{0.3em}
    \textbf{Akar penyebab}: TF-IDF tidak mengenal \textit{sinonim} --- ``mobil'' $\neq$ ``kendaraan'', ``mogok'' $\neq$ ``rusak''.
\end{frame}

\begin{frame}
    \frametitle{Kelemahan TF-IDF \#2}
    \framesubtitle{Tidak Paham Konteks}
    \small
    Satu kata bisa punya makna \textbf{berbeda} tergantung konteks kalimat.
    \vspace{0.5em}

    \begin{columns}[T]
        \column{0.47\textwidth}
        \begin{tcolorbox}[colback=nyublue!8, colframe=nyublue!50, title=Konteks Keuangan, fonttitle=\small\bfseries]
            \texttt{bank tempat menyimpan uang}
        \end{tcolorbox}
        \column{0.47\textwidth}
        \begin{tcolorbox}[colback=nyublue!8, colframe=nyublue!50, title=Konteks Alam, fonttitle=\small\bfseries]
            \texttt{bank sungai yang curam}
        \end{tcolorbox}
    \end{columns}
    \vspace{0.5em}
    \begin{alertblock}{Masalah}
        Kata \textbf{``bank''} mendapat skor TF-IDF yang \textbf{sama persis} di kedua kalimat, padahal artinya berbeda sama sekali.
    \end{alertblock}
    \vspace{0.3em}
    TF-IDF memperlakukan token sebagai simbol tanpa makna --- kata dikodekan sebagai angka, tidak pernah diinterpretasi berdasarkan lingkungannya.
\end{frame}

\begin{frame}
    \frametitle{Kelemahan TF-IDF \#3}
    \framesubtitle{Sparse \& Boros Memori}
    \small
    Ukuran vektor TF-IDF = ukuran \textit{vocabulary} seluruh corpus.
    \vspace{0.4em}
    \begin{itemize}
        \item Vocabulary 50.000 kata $\Rightarrow$ setiap dokumen menjadi \textbf{vektor sepanjang 50.000}.
        \item Mayoritas nilainya \textbf{nol} karena kata-kata tersebut tidak muncul di dokumen itu.
    \end{itemize}
    \vspace{0.3em}
    \begin{tcolorbox}[colback=gray!10, colframe=gray!40]
        \texttt{\small [0, 0, 0, 0, 0.3, 0, 0, 0, 0, 0, 0.5, 0, 0, \ldots]}\\[0.15em]
        \texttt{\small \hspace{10.2em}$\uparrow$ 99\% nilainya nol}
    \end{tcolorbox}
    \vspace{0.3em}
    \begin{exampleblock}{Dampak Praktis}
        Dataset 100.000 dokumen $\times$ 50.000 fitur = \textbf{5 miliar sel} --- sebagian besar kosong.
        Dataset besar $\Rightarrow$ memori sistem bisa habis.
    \end{exampleblock}
\end{frame}

\begin{frame}
    \frametitle{Kelemahan TF-IDF \#4}
    \framesubtitle{Tidak Paham Urutan Kata \& Out-of-Vocabulary}
    \small
    \textbf{(a) Urutan kata diabaikan sepenuhnya} (\textit{bag of words}):
    \vspace{0.2em}
    \begin{columns}[T]
        \column{0.47\textwidth}
        \begin{tcolorbox}[colback=gray!10, colframe=gray!40]
            \texttt{anjing menggigit orang}
        \end{tcolorbox}
        \column{0.47\textwidth}
        \begin{tcolorbox}[colback=gray!10, colframe=gray!40]
            \texttt{orang menggigit anjing}
        \end{tcolorbox}
    \end{columns}
    \vspace{0.2em}
    Kata sama, frekuensi sama $\Rightarrow$ \textbf{vektor identik} --- padahal maknanya sangat berbeda.
    \vspace{0.5em}

    \textbf{(b) Out-of-Vocabulary (OOV)} --- vocabulary dibekukan setelah training:
    \vspace{0.2em}
    \begin{tcolorbox}[colback=gray!10, colframe=gray!40]
        \texttt{\small Vocab: ["kopi", "teh", "susu"]}\\[0.15em]
        \texttt{\small Dokumen baru: "saya suka americano"}\\
        \texttt{\small \hspace{12.3em}$\uparrow$ tidak dikenal $\Rightarrow$ diabaikan}
    \end{tcolorbox}
\end{frame}

\begin{frame}
    \frametitle{Ringkasan: Kelemahan TF-IDF}
    \framesubtitle{Mengapa kita perlu pendekatan yang lebih canggih}
    \small
    \begin{center}
    \begin{tabular}{ll}
        \toprule
        \textbf{Kelemahan} & \textbf{Akibat} \\
        \midrule
        Tidak paham sinonim   & Kalimat mirip dianggap tidak berkaitan \\
        Tidak paham konteks   & Kata ambigu salah diinterpretasi \\
        Sparse vector         & Boros memori, komputasi lambat \\
        Abaikan urutan kata   & Makna struktural hilang \\
        Out-of-Vocabulary     & Kata baru diabaikan total \\
        \bottomrule
    \end{tabular}
    \end{center}
    \vspace{0.5em}
    \begin{tcolorbox}[colback=nyublue!10, colframe=nyublue, boxrule=0.8mm, arc=2mm, halign=center]
        Semua kelemahan ini dijawab oleh \textbf{Word Embedding} ---
        representasi \textit{dense} yang paham sinonim, urutan, dan konteks.
    \end{tcolorbox}
\end{frame}

    \end{columns}

    \vspace{0.6em}
    \begin{center}

% --- 2.1 Neuron Buatan ---
\begin{frame}
    \frametitle{Neuron: Unit Terkecil Deep Learning}
    \framesubtitle{Meniru cara kerja sel saraf di otak}
    \small
    Bayangkan neuron seperti \textbf{saklar cerdas}: ia menerima beberapa sinyal,
    menimbang-nimbangnya, lalu memutuskan apakah harus "menyala" atau tidak.

    \vspace{0.3em}
    \begin{center}
    \begin{tikzpicture}[
        node distance=0.4cm and 1.1cm,
        inp/.style={circle, draw=nyubluedarker, fill=nyublue!15, minimum size=0.7cm, font=\footnotesize},
        neu/.style={circle, draw=nyupurple, fill=nyupurple!15, minimum size=1cm, font=\footnotesize, align=center},
        outnd/.style={circle, draw=nyuorange!80!black, fill=nyuorange!15, minimum size=0.7cm, font=\footnotesize},
        arr/.style={-{Stealth[length=4.5pt]}, thick}
    ]
        \node[inp] (x1) {$x_1$};
        \node[inp, below=of x1] (x2) {$x_2$};
        \node[inp, below=of x2] (x3) {$x_3$};

        \node[neu, right=1.4cm of x2] (n) {$\Sigma$\\+\\$f$};

        \node[outnd, right=1.4cm of n] (y) {$y$};

        \draw[arr, color=nyubluedarker] (x1) -- node[above, font=\tiny, sloped]{$w_1$} (n);
        \draw[arr, color=nyubluedarker] (x2) -- node[above, font=\tiny]{$w_2$} (n);
        \draw[arr, color=nyubluedarker] (x3) -- node[above, font=\tiny, sloped]{$w_3$} (n);
        \draw[arr, color=nyupurple] (n) -- (y);

        % bias annotation
        \node[above=0.15cm of n, font=\tiny, color=nyupurple] {$+\,b$};
    \end{tikzpicture}
    \end{center}

    \vspace{0.2em}
    \footnotesize
    \begin{itemize}
        \item $x_1, x_2, x_3$ = \textbf{input}, $\;$ $w_1, w_2, w_3$ = \textbf{weight} (bobot kepentingan)
        \item $b$ = \textbf{bias} (pergeseran ambang), $\;$ $f$ = \textbf{fungsi aktivasi}
        \item Rumus: $y = f\bigl(w_1 x_1 + w_2 x_2 + w_3 x_3 + b\bigr)$
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Weight \& Bias: Parameter yang Dipelajari}
    \framesubtitle{Model belajar = model menyetel angka-angka ini}
    \footnotesize
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{block}{Weight ($w$)}
            \begin{itemize}
                \item Menentukan \textbf{seberapa besar pengaruh} suatu input
                \item $w$ besar $\rightarrow$ sangat berpengaruh
                \item $w$ negatif $\rightarrow$ bersifat "menghambat"
            \end{itemize}
        \end{block}

        \column{0.48\textwidth}
        \begin{exampleblock}{Bias ($b$)}
            \begin{itemize}
                \item Menggeser titik keputusan neuron
                \item Neuron bisa aktif meski semua input = 0
                \item Seperti $b$ dalam $y = mx + b$
            \end{itemize}
        \end{exampleblock}
    \end{columns}

    \vspace{0.3em}
    \begin{alertblock}{Inisialisasi Weight}
        Dimulai dari nilai acak kecil. Metode umum: \textbf{Xavier} atau \textbf{He initialization}. Nilai disesuaikan bertahap selama training.
    \end{alertblock}
\end{frame}

% --- 2.2 Fungsi Aktivasi ---
\begin{frame}
    \frametitle{Fungsi Aktivasi}
    \framesubtitle{Memberi kemampuan "berpikir non-linear" pada neuron}
    \small
    Tanpa fungsi aktivasi, seluruh jaringan hanya melakukan operasi linear.

    \vspace{0.2em}
    \begin{center}
    {\footnotesize
    \begin{tabular}{llp{5.2cm}}
        \toprule
        \textbf{Fungsi} & \textbf{Rentang Output} & \textbf{Kapan digunakan} \\
        \midrule
        \textbf{ReLU}    & $[0, \infty)$  & Hidden layer — paling umum, komputasi cepat \\
        \textbf{Sigmoid} & $(0, 1)$       & Output layer klasifikasi \textit{biner} \\
        \textbf{Tanh}    & $(-1, 1)$      & Hidden layer RNN / LSTM — mengatasi batasan Sigmoid \\
        \textbf{Softmax} & $(0, 1)$, jumlah = 1 & Output layer klasifikasi \textit{multikelas} \\
        \bottomrule
    \end{tabular}}
    \end{center}

    \vspace{0.15em}
    \begin{exampleblock}{Analogi ReLU}
        ReLU ($\max(0, x)$) seperti lampu yang hanya menyala saat ada listrik positif — nilai negatif $\rightarrow$ tetap 0.
    \end{exampleblock}
\end{frame}

% --- 2.4 Loss Function ---
\begin{frame}
    \frametitle{Loss Function: Mengukur Seberapa Salah Model}
    \framesubtitle{Sinyal yang memberi tahu model "kamu masih keliru"}
    \small
    \textbf{Loss function} mengukur selisih prediksi model vs.~jawaban benar.
    Makin kecil \textit{loss}, makin bagus.

    \begin{exampleblock}{Cross-Entropy Loss — Klasifikasi Multikelas}
        \[
            \mathcal{L} = -\sum_{c=1}^{C} y_c \cdot \log(\hat{p}_c)
        \]
        \footnotesize
        \begin{itemize}
            \item $y_c = 1$ jika kelas $c$ benar, 0 jika tidak
            \item $\hat{p}_c$ = probabilitas prediksi untuk kelas $c$ (output Softmax)
        \end{itemize}
    \end{exampleblock}

    \vspace{0.15em}
    \begin{alertblock}{Contoh}
        \footnotesize Jawaban: \texttt{Anxiety}. Prediksi \texttt{Anxiety} 90\% $\rightarrow$ loss kecil. Prediksi \texttt{Normal} 90\% $\rightarrow$ loss besar.
    \end{alertblock}
\end{frame}

% --- 2.6 Overfitting & Regularisasi ---
\begin{frame}
    \frametitle{Overfitting: Model Terlalu "Hafal"}
    \framesubtitle{Masalah klasik yang sering muncul saat melatih model}
    \small
    \textbf{Overfitting} terjadi ketika model terlalu fokus menghafal data latih,
    sehingga performa di data baru (data uji) justru buruk.

    \vspace{0.4em}
    \begin{center}
    \begin{tikzpicture}[scale=0.78]
        % axes
        \draw[->] (0,0) -- (5.5,0) node[right,font=\footnotesize]{Epoch};
        \draw[->] (0,0) -- (0,2.8) node[above,font=\footnotesize]{Loss};

        % train loss (goes down smoothly)
        \draw[thick, color=nyubluedarker, domain=0.3:5, samples=50]
            plot (\x, {2.2*exp(-0.7*\x)+0.15});
        \node[font=\tiny, color=nyubluedarker] at (4.5,0.4) {Train Loss};

        % val loss (goes down then up)
        \draw[thick, color=nyuorange!80!black, domain=0.3:5, samples=50]
            plot (\x, {2.2*exp(-0.7*\x)+0.15 + 0.5*max(0,\x-2.5)^1.2});
        \node[font=\tiny, color=nyuorange!80!black] at (4.2,2.0) {Val Loss};

        % overfitting marker
        \draw[dashed, gray] (2.5,0) -- (2.5,2.5);
        \node[font=\tiny, gray, above] at (2.5,2.5) {mulai \textit{overfit}};
    \end{tikzpicture}
    \end{center}

    \vspace{0.1em}
    \footnotesize Saat \textit{val loss} mulai naik sementara \textit{train loss} terus turun, itulah tanda overfitting.
\end{frame}

\begin{frame}
    \frametitle{Regularisasi: Cara Mencegah Overfitting}
    \framesubtitle{Tiga teknik yang paling sering digunakan}
    \small
    \begin{columns}[T]
        \column{0.32\textwidth}
        \begin{block}{Dropout}
            Saat training, sebagian neuron \textbf{dimatikan secara acak} di setiap iterasi.\\[0.3em]
            Memaksa model tidak bergantung pada neuron tertentu saja.\\[0.3em]
            \footnotesize Contoh: \texttt{Dropout(0.3)} $\rightarrow$ 30\% neuron dimatikan tiap langkah.
        \end{block}

        \column{0.32\textwidth}
        \begin{exampleblock}{L2 Regularization}
            Menambahkan \textbf{penalti} pada nilai weight yang terlalu besar ke dalam loss function.\\[0.3em]
            Weight besar $\rightarrow$ model mengandalkan fitur tertentu secara berlebihan.\\[0.3em]
            \footnotesize Disebut juga \textit{weight decay}.
        \end{exampleblock}

        \column{0.32\textwidth}
        \begin{alertblock}{Early Stopping}
            Hentikan training \textbf{saat val loss mulai naik}, bukan saat mencapai epoch maksimum.\\[0.3em]
            \footnotesize Parameter: \texttt{patience=3} $\rightarrow$ training berhenti jika val loss tidak membaik selama 3 epoch berturut-turut.
        \end{alertblock}
    \end{columns}
\end{frame}

% --- WE 1: Masalah one-hot ---
\begin{frame}[shrink=5]
    \frametitle{Dari Kata ke Angka}
    \framesubtitle{One-Hot Encoding: cara paling polos}
    \small
    Komputer tidak bisa membaca teks --- kita harus mengubah setiap kata menjadi \textbf{angka} dulu.
    Cara paling sederhana: \textbf{One-Hot Encoding}. Vocab = \texttt{[anjing, kucing, ikan, burung]}
    \vspace{0.15em}
    \begin{center}
    \footnotesize
    \begin{tabular}{lcccc}
        \toprule
        Kata & anjing & kucing & ikan & burung \\
        \midrule
        \texttt{anjing} & \textbf{1} & 0 & 0 & 0 \\
        \texttt{kucing} & 0 & \textbf{1} & 0 & 0 \\
        \texttt{ikan}   & 0 & 0 & \textbf{1} & 0 \\
        \texttt{burung} & 0 & 0 & 0 & \textbf{1} \\
        \bottomrule
    \end{tabular}
    \end{center}
    \vspace{0.1em}
    \begin{alertblock}{Masalah}
        \footnotesize
        \begin{itemize}
            \item \textbf{Semua kata sama jauhnya}: jarak \texttt{anjing}--\texttt{kucing} = jarak \texttt{anjing}--\texttt{ikan}, padahal keduanya sama-sama hewan peliharaan.
            \item \textbf{Sparse}: 50.000 kata $\Rightarrow$ vektor 50.000 elemen, hampir semuanya nol.
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}

    \small
    Bayangkan kamu bisa \textbf{memetakan} setiap kata ke sebuah titik di ruang koordinat.\\

% --- WE 3: Vektor multi-dimensi ---
\begin{frame}[shrink=6]
    \frametitle{Kata sebagai Koordinat Multi-Dimensi}
    \framesubtitle{Tiap dimensi menangkap satu aspek makna}
    \small
    Setiap kata direpresentasikan sebagai \textbf{daftar angka} (vektor):
    \vspace{0.1em}
    \begin{center}
    \footnotesize
    \begin{tabular}{lrrrrrr}
        \toprule
        Kata & $d_1$ & $d_2$ & $d_3$ & $d_4$ & $d_5$ & \ldots \\
        \midrule
        \texttt{raja}   &  0.82 & -0.41 &  0.03 &  0.71 &  0.55 & \ldots \\
        \texttt{ratu}   &  0.79 & -0.38 &  0.02 & -0.69 &  0.52 & \ldots \\
        \texttt{pria}   &  0.45 & -0.90 &  0.01 &  0.66 &  0.20 & \ldots \\
        \texttt{wanita} &  0.43 & -0.87 &  0.00 & -0.64 &  0.21 & \ldots \\
        \texttt{kopi}   & -0.10 &  0.12 &  0.89 &  0.05 & -0.30 & \ldots \\
        \bottomrule
    \end{tabular}
    \end{center}
    \vspace{0.1em}
    \begin{itemize}
        \item Dimensi-dimensi ini \textbf{tidak diberi nama} oleh manusia --- model yang menemukan sendiri.
        \item Bisa saja $d_4$ menangkap ``jenis kelamin'', $d_1$ menangkap ``jabatan'' --- kita tidak tahu persis.
        \item \texttt{raja} dan \texttt{ratu} hampir identik di semua dimensi, kecuali $d_4$ yang berlawanan tanda.
    \end{itemize}

% --- WE 4: Word2Vec ---
\begin{frame}
    \frametitle{Word2Vec: Cara Melatih Embedding}
    \framesubtitle{"Kamu dinilai dari teman-temanmu"}
    \small
    \textbf{Ide utama Word2Vec}: sebuah kata bisa dipahami maknanya dari kata-kata \textit{di sekitarnya}.

    \vspace{0.15em}
    \begin{tcolorbox}[colback=nyublue!8, colframe=nyublue!50]
        \centering
        \textit{``Saya tadi minum \underline{\hspace{1.2cm}} di warung kopi.''}\\
        \small Kata apa yang masuk akal? $\Rightarrow$ \textbf{kopi}, \textbf{teh}, \textbf{susu} --- bukan \textbf{mobil} atau \textbf{gunung}.
    \end{tcolorbox}
    \vspace{0.15em}
    Word2Vec melatih model untuk \textbf{menebak kata yang hilang} dari konteks kalimat.
    Kata-kata dengan konteks serupa otomatis mendapat \textbf{vektor yang mirip}.
    \vspace{0.2em}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \textbf{CBOW} (\textit{Continuous Bag of Words})\\
        \small Konteks $\rightarrow$ tebak kata tengah\\
        \texttt{[minum, di, warung]} $\rightarrow$ \texttt{kopi}
        \column{0.48\textwidth}
        \textbf{Skip-Gram}\\
        \small Kata tengah $\rightarrow$ tebak konteks\\
        \texttt{kopi} $\rightarrow$ \texttt{[minum, di, warung]}
    \end{columns}
    \vspace{0.2em}
    \begin{exampleblock}{Hasil Training}
        Model tidak pernah diajarkan ``kopi dan teh itu mirip'' --- ia \textbf{menemukan sendiri} lewat jutaan kalimat.
    \end{exampleblock}
\end{frame}

% --- WE 6: Vector arithmetic ---
\begin{frame}
    \frametitle{Sifat Ajaib Word Embedding}
    \framesubtitle{Matematika dengan makna kata}
    \small
    Karena kata direpresentasikan sebagai vektor, kita bisa melakukan \textbf{operasi matematika} pada \textit{makna}:

    \vspace{0.15em}
    \begin{center}
    \begin{tcolorbox}[colback=nyublue!10, colframe=nyublue, boxrule=0.8mm, width=0.78\textwidth, halign=center]
        \large
        $\vec{\texttt{raja}} - \vec{\texttt{pria}} + \vec{\texttt{wanita}} \approx \vec{\texttt{ratu}}$
    \end{tcolorbox}
    \end{center}
    \vspace{0.1em}
    \begin{center}
    \begin{tcolorbox}[colback=nyublue!10, colframe=nyublue, boxrule=0.8mm, width=0.78\textwidth, halign=center]
        \large
        $\vec{\texttt{Paris}} - \vec{\texttt{Prancis}} + \vec{\texttt{Indonesia}} \approx \vec{\texttt{Jakarta}}$
    \end{tcolorbox}
    \end{center}
    \vspace{0.15em}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \textbf{Apa artinya ini?}
        \begin{itemize}
            \item Embedding menangkap \textbf{relasi} antar konsep, bukan hanya kata.
            \item Konsep ``jenis kelamin'' dan ``ibu kota'' bisa dihitung seperti aritmetika biasa.
        \end{itemize}
        \column{0.48\textwidth}
        \textbf{Bagaimana ini terjadi?}\\[0.2em]
        Murni dari jutaan kalimat --- model tidak pernah diajarkan konsep ``ibu kota'' secara eksplisit. Pola itu \textbf{muncul sendiri}.
    \end{columns}
\end{frame}

\section{Dari MLP ke Model Sekuensial}

% --- 3.1 Masalah MLP untuk Teks ---
\begin{frame}
    \frametitle{Masalah MLP untuk Teks}
    \framesubtitle{Kenapa jaringan biasa tidak cukup?}
    \small
    Bayangkan kamu diminta menebak makna kalimat ini:
    \begin{center}
        \textit{"Saya \textbf{tidak} baik-baik saja"}
    \end{center}
    vs.
    \begin{center}
        \textit{"Saya baik-baik saja, \textbf{tidak} ada yang salah"}
    \end{center}

    \vspace{0.4em}
    MLP melihat kedua kalimat hanya sebagai \textbf{kumpulan kata} — kata "tidak" dan "baik" ada di keduanya. Ia \tem{tidak peduli} urutan.

    \vspace{0.3em}
    \begin{alertblock}{Inti Masalah}
        Teks adalah data \textbf{berurutan} (\textit{sequential}). Makna sebuah kata bergantung pada kata sebelumnya.
    \end{alertblock}
\end{frame}

% --- 3.2 RNN: Ide Dasar ---
\begin{frame}
    \frametitle{RNN — Membaca Kata Satu per Satu}
    \framesubtitle{Seperti kamu membaca kalimat dari kiri ke kanan}
    \small
    \textbf{RNN} (\textit{Recurrent Neural Network}) membaca teks \textbf{kata demi kata}, sambil menyimpan "ingatan" dari kata yang sudah dibaca.

    \vspace{0.5em}
    \begin{center}

        arr/.style={-{Stealth[length=4.5pt]}, thick, color=nyupurple},
        harr/.style={-{Stealth[length=4.5pt]}, thick, color=nyuorange!80!black}
    ]
        \node[rnn] (r1) {RNN};
        \node[rnn, right=1.3cm of r1] (r2) {RNN};
        \node[rnn, right=1.3cm of r2] (r3) {RNN};
        \node[rnn, right=1.3cm of r3] (r4) {RNN};

        \node[inp, below=0.5cm of r1] (x1) {"Saya"};
        \node[inp, below=0.5cm of r2] (x2) {"tidak"};
        \node[inp, below=0.5cm of r3] (x3) {"baik"};
        \node[inp, below=0.5cm of r4] (x4) {"saja"};

    $h_t$ = \textbf{hidden state} — semacam "catatan" yang diteruskan dari satu kata ke kata berikutnya.
\end{frame}

% --- 3.3 Hidden State sebagai Memori ---
\begin{frame}
    \frametitle{Hidden State: "Memori" RNN}
    \framesubtitle{Informasi yang dibawa dari kata ke kata}
    \small
    \begin{columns}[T]
        \column{0.52\textwidth}
        Setiap langkah, RNN menghitung hidden state baru:
        \[
            h_t = \tanh(W_h \cdot h_{t-1} + W_x \cdot x_t + b)
        \]
        \footnotesize
        \begin{itemize}
            \item $h_{t-1}$ = ingatan dari kata sebelumnya
            \item $x_t$ = kata saat ini (dalam bentuk vektor)
            \item Hasilnya $h_t$ diteruskan ke langkah berikutnya
        \end{itemize}

        \column{0.44\textwidth}
        \begin{exampleblock}{Analogi}
            Bayangkan kamu membaca novel sambil memegang \textbf{catatan kecil}. Setiap halaman, kamu update catatanmu. Di akhir bab, catatanmu merangkum semua yang terjadi.
        \end{exampleblock}
    \end{columns}
\end{frame}

% --- 3.4 Masalah Vanishing Gradient ---
\begin{frame}
    \frametitle{Masalah RNN: "Lupa" Teks Panjang}
    \framesubtitle{Vanishing Gradient — musuh utama RNN}
    \small
    RNN bekerja bagus untuk kalimat pendek. Tapi untuk teks panjang, ada masalah besar:

    \vspace{0.4em}
    \begin{center}
    \begin{tikzpicture}[

        }
        \foreach \i [evaluate=\i as \j using int(\i+1)] in {1,...,5} {
            \draw[arr, color=gray!\j0] (r\i)--(r\j);
        }
        \node[font=\tiny, below=3pt of r1, color=black] {kata ke-1};
        \node[font=\tiny, below=3pt of r6, color=gray!50] {kata ke-50};
        \draw[dashed, nyuorange!80!black, thick] (r1.north)++(0,0.3) -- ++(0,-0.9);

    \footnotesize Akibatnya: RNN "lupa" kata-kata yang jauh di awal kalimat.
\end{frame}

% --- 3.5 LSTM ---
\begin{frame}[shrink=3]
    \frametitle{LSTM: Solusi untuk Ingatan Jangka Panjang}
    \framesubtitle{Long Short-Term Memory}
    \small
    \textbf{LSTM} menambahkan mekanisme \textbf{gerbang} (\textit{gate}) — seperti filter yang memutuskan:
    \begin{itemize}
        \item Apa yang perlu \textbf{diingat} dari masa lalu
        \item Apa yang boleh \textbf{dilupakan}
        \item Apa yang perlu \textbf{dikeluarkan} sebagai output
    \end{itemize}

    \vspace{0.2em}
    \begin{center}
    \begin{tikzpicture}[
        gate/.style={draw=nyupurple, fill=nyupurple!12, rounded corners=3pt,

    ]

        \node[gate, right=0.8cm of fg] (ig) {Input\\Gate};
        \node[gate, right=0.8cm of ig] (og) {Output\\Gate};

        \draw[carr, line width=1.2pt] (-0.8, 1.0) -- (fg.north west |- 0,1.0)
            -- (ig.north |- 0,1.0) -- (og.north east |- 0,1.0) -- ++(0.8,0)
            node[right, font=\tiny, color=nyuorange!80!black]{$c_t$ (memori panjang)};

        \draw[arr] (fg.north) -- (fg.north |- 0,1.0);

    \end{center}

    \vspace{0.1em}
    \footnotesize
    \begin{exampleblock}{Analogi}
        Seperti membaca buku dengan dua catatan: \textbf{catatan panjang} (ingatan penting) dan \textbf{catatan kerja} (konteks saat ini).
    \end{exampleblock}
\end{frame}

% --- 3.6 BiLSTM ---

    \small
    Dalam bahasa, konteks datang dari \textbf{dua arah}. Contoh:
    \begin{center}

    \end{center}

% --- 4.2 Solusi: Attention ---
\begin{frame}
    \frametitle{Solusi: Beri Model Kemampuan "Menyorot"}
    \framesubtitle{Ide dasar Attention Mechanism}
    \small
    Daripada hanya memakai \textit{last hidden state}, bagaimana kalau model boleh \textbf{melihat semua hidden state} sekaligus dan memilih mana yang paling penting?

    \vspace{0.4em}

        arr/.style={-{Stealth[length=4pt]}, thick}
    ]
        \node[rnn] (h1) {$h_1$};
        \node[rnn, right=0.8cm of h1] (h2) {$h_2$};
        \node[rnn, right=0.8cm of h2] (h3) {$h_3$};
        \node[rnn, right=0.8cm of h3] (h4) {$h_4$};
        \node[rnn, right=0.8cm of h4] (h5) {$h_5$};

        \node[att, below=0.85cm of h3] (attn) {Attention Layer};
        \node[cls, right=1.0cm of attn] (fc) {Classifier};

% --- 4.3 Step-by-step Attention ---
\begin{frame}
    \frametitle{Cara Kerja Attention: Step by Step}
    \framesubtitle{Bahdanau Attention — langkah demi langkah}
    \small
    \begin{enumerate}\setlength{\itemsep}{0.05em}
        \item \textbf{BiLSTM menghasilkan hidden state} tiap kata: $h_1, h_2, \ldots, h_T$

        \item \textbf{Hitung skor} — seberapa penting kata ini?
        $\displaystyle e_t = \tanh(W \cdot h_t + b)$

        \item \textbf{Normalisasi} dengan Softmax (total = 1):
        $\displaystyle \alpha_t = \dfrac{\exp(e_t)}{\sum_{k=1}^{T} \exp(e_k)}$

        \item \textbf{Buat context vector} — rata-rata tertimbang:
        $\displaystyle c = \sum_{t=1}^{T} \alpha_t \cdot h_t$
    \end{enumerate}

    \vspace{0.2em}
    \begin{exampleblock}{}
        \footnotesize
        $c$ inilah yang diteruskan ke classifier — merangkum seluruh kalimat dengan \textbf{penekanan lebih pada kata yang penting}.
    \end{exampleblock}
\end{frame}

            \begin{enumerate}\setlength{\itemsep}{0pt}

                \item Buat \textit{context vector} $c$
                \item Masukkan $c$ ke FC + Softmax
            \end{enumerate}

            \textcolor{nyubluedarker}{Lebih akurat \textbf{dan} bisa dijelaskan.}
        \end{exampleblock}

\begin{frame}

    \small

    \vspace{0.3em}

\begin{frame}

    \framesubtitle{Inilah kekuatan utama Transformer}
    \small
    Kalimat: \textit{"The \textbf{bank} by the \textbf{river} is flooded"}\\[0.3em]
    Kata \textit{bank} perlu "tahu" bahwa ia berada di dekat \textit{river} agar model memilih makna \textbf{tepi sungai}, bukan \textbf{lembaga keuangan}.

    \vspace{0.3em}
    \begin{center}
    \begin{tikzpicture}[
        tok/.style={draw=nyubluedarker, fill=nyublue!12, rounded corners=2pt,

        htok/.style={draw=nyupurple, fill=nyupurple!15, rounded corners=2pt,
                    minimum height=0.55cm, minimum width=0.95cm, font=\scriptsize, align=center},
        arr/.style={-{Stealth[length=3.5pt]}}
    ]
        \node[htok] (bank)  at (0,0)   {bank};

        \node[tok]  (is)    at (5.2,0)  {is};

        \text{Attention}(Q,K,V) \;=\; \text{softmax}\!\left(\frac{QK^\top}{\sqrt{d_k}}\right)V
    \]
    \footnotesize\textcolor{gray}{Sederhananya: Q = "apa yang saya cari?", \; K = "apa yang tersedia?", \; V = "informasi yang dibagi".}
\end{frame}

\begin{frame}

    \framesubtitle{Bidirectional Encoder Representations from Transformers}
    \small
    \textbf{BERT} dilatih Google pada hampir seluruh Wikipedia + jutaan buku.\\[-0.1em]
    Dua hal unik yang membuatnya revolusioner:

    \vspace{0.2em}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{block}{1. Baca Dua Arah (Bidirectional)}
            \footnotesize
            Model sebelumnya (GPT) membaca kiri ke kanan saja.\\[0.2em]
            BERT membaca \textbf{kiri dan kanan sekaligus}.
            \begin{center}
            \small\texttt{saya \underline{\hspace{0.7cm}} kopi}\\
            \small\textcolor{nyublue}{$\leftarrow$ dan $\rightarrow$ diperhatikan}
            \end{center}
        \end{block}

        \column{0.48\textwidth}
        \begin{block}{2. Masked Language Model}
            \footnotesize
            Saat pre-training, BERT \textbf{menyembunyikan} 15\% kata secara acak, lalu menebaknya dari konteks kiri dan kanan sekaligus.\\[0.2em]
            \texttt{saya [MASK] kopi} $\to$ \texttt{suka}
        \end{block}
    \end{columns}

    \vspace{0.2em}
    \begin{exampleblock}{Hasil}
        \footnotesize
        BERT memecahkan rekor di 11 tugas NLP saat dirilis (2018). Masalahnya? \textbf{110 juta parameter}, butuh GPU besar, dan lambat untuk produksi.
    \end{exampleblock}
\end{frame}

% --- B2: Masalah BERT, lahirnya DistilBERT ---
\begin{frame}[shrink=5]
    \frametitle{Masalah BERT di Dunia Nyata}
    \framesubtitle{Terlalu besar, terlalu lambat}
    \small
    BERT sangat canggih --- tapi ada harga yang harus dibayar:

    \vspace{0.2em}
    \begin{center}
    \begin{tabular}{lcc}
        \toprule
        \textbf{Masalah} & \textbf{BERT-base} & \textbf{Dampak} \\
        \midrule
        Jumlah layer        & 12 layer Transformer & Lambat \\
        Parameter           & 110 juta             & File 440 MB \\
        RAM saat inference  & $\sim$1.5 GB         & Tidak bisa di HP \\
        Waktu per prediksi  & $\sim$5 ms           & Sulit real-time \\
        \bottomrule
    \end{tabular}
    \end{center}

    \vspace{0.2em}
    \begin{alertblock}{Pertanyaan}
        Bisakah kita buat model yang \textbf{hampir secerdas BERT} tapi jauh \textbf{lebih kecil dan cepat}?
    \end{alertblock}

    \vspace{0.2em}
    Jawabannya datang dari tim Hugging Face pada 2019:
    \begin{center}
        \large\textcolor{nyupurple}{\textbf{Knowledge Distillation $\to$ DistilBERT}}
    \end{center}
\end{frame}

% --- B3: Analogi Knowledge Distillation ---
\begin{frame}
    \frametitle{Knowledge Distillation: Murid Belajar dari Guru}
    \framesubtitle{Bukan meniru jawaban --- meniru cara berpikir}
    \small
    Bayangkan dua cara belajar:

    \vspace{0.3em}
    \begin{columns}[T]
        \column{0.47\textwidth}
        \begin{alertblock}{Cara Biasa (Hard Label)}
            \footnotesize
            Guru cuma bilang:\\[0.2em]
            \textit{``Kalimat ini = kelas \textbf{Depressed}.''}\\[0.3em]
            Murid hanya tahu jawaban akhir --- tidak tahu \textit{kenapa}.
        \end{alertblock}

        \column{0.47\textwidth}
        \begin{exampleblock}{Distillation (Soft Label)}
            \footnotesize
            Guru bilang:\\[0.2em]
            \textit{``Ini 70\% Depressed, 20\% Anxiety, 10\% Normal.''}\\[0.3em]
            Murid belajar dari \textbf{pola kepercayaan} guru --- jauh lebih kaya informasi.
        \end{exampleblock}
    \end{columns}

    \vspace{0.4em}
    \begin{center}
    \begin{tikzpicture}[
        box/.style={draw=nyubluedarker, fill=nyublue!10, rounded corners=3pt,
                    minimum height=0.8cm, text width=2.5cm, align=center, font=\footnotesize},
        sbox/.style={draw=nyupurple, fill=nyupurple!10, rounded corners=3pt,
                    minimum height=0.8cm, text width=2.5cm, align=center, font=\footnotesize},
        arr/.style={-{Stealth[length=5pt]}, thick}
    ]
        \node[box]  (bert)   {BERT\\{\tiny guru, 110M param}};
        \node[sbox, right=1.8cm of bert] (distil) {DistilBERT\\{\tiny murid, 67M param}};
        \draw[arr, color=nyupurple, dashed]
            (bert) -- node[above, font=\tiny]{soft labels} (distil);
        \node[font=\tiny, color=gray, below=5pt of bert] {sudah jadi, tidak dilatih ulang};
        \node[font=\tiny, color=nyupurple, below=5pt of distil] {dilatih meniru BERT};
    \end{tikzpicture}
    \end{center}
\end{frame}

% --- B4: Cara kerja distilasi teknis ---
\begin{frame}[shrink=5]
    \frametitle{Cara Kerja Distilasi: 3 Sinyal Sekaligus}
    \framesubtitle{DistilBERT dilatih dengan tiga sumber belajar}
    \small
    Saat melatih DistilBERT, loss gabungan dari \textbf{tiga sinyal}:

    \vspace{0.2em}
    \begin{enumerate}
        \item \textbf{Distillation Loss} --- seberapa mirip distribusi probabilitas output DistilBERT dengan BERT.\\
              \textcolor{gray}{\footnotesize Pakai \textit{temperature softmax}: output BERT ``dilunakkan'' agar perbedaan kecil antar kelas tetap terlihat.}

        \vspace{0.2em}
        \item \textbf{MLM Loss} --- DistilBERT tetap dilatih menebak kata yang disembunyikan (\textit{masked language model}) seperti BERT.

        \vspace{0.2em}
        \item \textbf{Cosine Embedding Loss} --- vektor tersembunyi DistilBERT diarahkan sedekat mungkin dengan vektor BERT.
    \end{enumerate}

    \vspace{0.2em}
    \begin{tcolorbox}[colback=nyublue!8, colframe=nyublue, boxrule=0.7mm, arc=2mm]
        \footnotesize
        \textbf{Hasilnya}: DistilBERT bukan sekadar ``BERT yang dipotong'' --- ia punya \textit{pemahaman bahasa} yang hampir setara, hanya lebih kompak.
    \end{tcolorbox}
\end{frame}

% --- B5: Perbandingan arsitektur ---
\begin{frame}
    \frametitle{Arsitektur: BERT vs DistilBERT}
    \framesubtitle{Setengah layer, akurasi tetap terjaga}
    \small
    \begin{columns}[T]
        \column{0.58\textwidth}
        \footnotesize
        \setlength{\tabcolsep}{5pt}
        \begin{tabular}{lcc}
            \toprule
            \textbf{Komponen}        & \textbf{BERT-base} & \textbf{DistilBERT} \\
            \midrule
            Transformer layers       & 12                 & \textbf{6} \\
            Hidden size              & 768                & 768 \\
            Attention heads          & 12                 & 12 \\
            Parameter                & 110 juta           & \textbf{67 juta} \\
            Token type embeddings    & Ada                & \textbf{Dihapus} \\
            Pooler layer             & Ada                & \textbf{Dihapus} \\
            \midrule
            Inference speed          & 1$\times$          & \textbf{1.6$\times$} \\
            GLUE score               & 79.6               & \textbf{77.0} ($-3.5\%$) \\
            \bottomrule
        \end{tabular}

        \column{0.38\textwidth}
        \begin{exampleblock}{Intuisi}
            \footnotesize
            Bukan semua layer itu sama pentingnya. DistilBERT membuktikan bahwa \textbf{6 layer yang terlatih dengan benar} bisa sehebat 12 layer biasa.
        \end{exampleblock}

        \vspace{0.4em}
        \footnotesize
        Perubahan utama: layer 12 $\to$ 6, token type embedding \& pooler \textbf{dihapus}, hidden size \& head \textbf{dipertahankan}.
    \end{columns}
\end{frame}

\begin{frame}[shrink=5]
    \frametitle{Ringkasan: Pilih BERT atau DistilBERT?}
    \framesubtitle{Tidak ada yang selalu lebih baik --- tergantung konteks}
    \small
    \begin{center}
    \setlength{\tabcolsep}{4pt}
    \begin{tabular}{lcc}
        \toprule
        \textbf{Situasi}                        & \textbf{BERT} & \textbf{DistilBERT} \\
        \midrule
        Butuh akurasi absolut tertinggi         & \checkmark    & \\
        Deployment di server terbatas           &               & \checkmark \\
        Inferensi real-time ($<$5ms)            &               & \checkmark \\
        Fine-tuning cepat dengan RAM kecil     &               & \checkmark \\
        Riset / benchmark akademis              & \checkmark    & \\
        Aplikasi mobile / edge device           &               & \checkmark \\
        \bottomrule
    \end{tabular}
    \end{center}

    \vspace{0.35em}
    \begin{tcolorbox}[colback=nyublue!10, colframe=nyublue, boxrule=0.8mm, arc=2mm]
        \small
        \textbf{Untuk workshop ini}: DistilBERT dipilih karena lebih \textbf{ramah resource}, bisa berjalan di laptop biasa, dan perbedaan akurasinya dengan BERT sangat kecil ($<4\%$) untuk tugas klasifikasi teks.
    \end{tcolorbox}

\end{frame}
